🔹 Ansible — Install & Enable Nginx
- name: Install and start Nginx on Gitpush
  hosts: all
  become: yes

  tasks:
    - name: Install nginx
      yum:
        name: nginx
        state: present

    - name: Enable nginx
      service:
        name: nginx
        state: started
        enabled: yes
